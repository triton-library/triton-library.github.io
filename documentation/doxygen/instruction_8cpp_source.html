395  bool Instruction::isReadFrom(const triton::arch::OperandWrapper& target) const {
396  switch(target.getType()) {
397 
398  case triton::arch::OP_IMM:
399  for (auto&& pair : this->readImmediates) {
400  if (pair.first == target.getConstImmediate())
401  return true;
402  }
403  break;
404 
405  case triton::arch::OP_MEM:
406  for (auto&& pair : this->loadAccess) {
407  const triton::arch::MemoryAccess& m1 = pair.first;
408  const triton::arch::MemoryAccess& m2 = target.getConstMemory();
409 
410  if (m1.isOverlapWith(m2))
411  return true;
412  }
413  break;
414 
415  case triton::arch::OP_REG:
416  for (auto&& pair : this->readRegisters) {
417  const triton::arch::Register& r1 = pair.first;
418  const triton::arch::Register& r2 = target.getConstRegister();
419 
420  if (r1.isOverlapWith(r2))
421  return true;
422  }
423  break;
424 
425  default:
426  throw triton::exceptions::Instruction("Instruction::isReadFrom(): Invalid type operand.");
427  }
428 
429  return false;
430  }
