60  bool IrBuilder::buildSemantics(triton::arch::Instruction& inst) {
61  triton::arch::architecture_e arch = this->architecture->getArchitecture();
62  bool ret = false;
63 
64  if (arch == triton::arch::ARCH_INVALID)
65  throw triton::exceptions::IrBuilder("IrBuilder::buildSemantics(): You must define an architecture.");
66 
67  /* Initialize the target address of memory operands */
68  for (auto& operand : inst.operands) {
69  if (operand.getType() == triton::arch::OP_MEM) {
70  this->symbolicEngine->initLeaAst(operand.getMemory());
71  }
72  }
73 
74  /* Pre IR processing */
75  this->preIrInit(inst);
76 
77  /* Processing */
78  switch (arch) {
79  case triton::arch::ARCH_AARCH64:
80  ret = this->aarch64Isa->buildSemantics(inst);
81  break;
82 
83  case triton::arch::ARCH_ARM32:
84  ret = this->arm32Isa->buildSemantics(inst);
85  break;
86 
87  case triton::arch::ARCH_X86:
88  case triton::arch::ARCH_X86_64:
89  ret = this->x86Isa->buildSemantics(inst);
90  break;
91 
92  default:
93  throw triton::exceptions::IrBuilder("IrBuilder::buildSemantics(): Architecture not supported.");
94  break;
95  }
96 
97  /* Post IR processing */
98  this->postIrInit(inst);
99 
100  return ret;
101  }

104  void IrBuilder::preIrInit(triton::arch::Instruction& inst) {
105  /* Clear previous expressions if exist */
106  inst.symbolicExpressions.clear();
107 
108  /* Clear implicit and explicit previous semantics */
109  inst.getLoadAccess().clear();
110  inst.getReadRegisters().clear();

122  void IrBuilder::postIrInit(triton::arch::Instruction& inst) {
123  std::vector<triton::engines::symbolic::SharedSymbolicExpression> newVector;
124 
125  /* Set the taint */
126  inst.setTaint();
127 
128  /*
129  * If the symbolic engine is defined to process symbolic
130  * execution only on symbolized expressions, we delete all
131  * concrete expressions and their AST nodes.
132  */
133  if (this->modes->isModeEnabled(triton::modes::ONLY_ON_SYMBOLIZED)) {
134  /* Clear memory operands */
135  this->collectUnsymbolizedNodes(inst.operands);
136 
137  /* Clear implicit and explicit semantics - MEM */
138  this->collectUnsymbolizedNodes(inst.getLoadAccess());
139 
140  /* Clear implicit and explicit semantics - REG */
141  this->collectUnsymbolizedNodes(inst.getReadRegisters());
142 
143  /* Clear implicit and explicit semantics - IMM */
144  this->collectUnsymbolizedNodes(inst.getReadImmediates());
145 
146  /* Clear implicit and explicit semantics - MEM */
147  this->collectUnsymbolizedNodes(inst.getStoreAccess());
148 
149  /* Clear implicit and explicit semantics - REG */
150  this->collectUnsymbolizedNodes(inst.getWrittenRegisters());
151 
152  /* Clear symbolic expressions */
153  for (const auto& se : inst.symbolicExpressions) {
154  if (se->isSymbolized() == false) {
155  this->symbolicEngine->removeSymbolicExpression(se);
156  }
157  else
158  newVector.push_back(se);
159  }
160  inst.symbolicExpressions = newVector;
161  }
162 
163  /*
164  * If the symbolic engine is defined to process symbolic
165  * execution only on tainted instructions, we delete all
166  * expressions untainted and their AST nodes.
167  */
168  else if (this->modes->isModeEnabled(triton::modes::ONLY_ON_TAINTED) && !inst.isTainted()) {
169  /* Memory operands */
170  this->collectNodes(inst.operands);
171 
172  /* Implicit and explicit semantics - MEM */
173  this->collectNodes(inst.getLoadAccess());
174 
175  /* Implicit and explicit semantics - REG */
176  this->collectNodes(inst.getReadRegisters());
177 
178  /* Implicit and explicit semantics - IMM */
179  this->collectNodes(inst.getReadImmediates());
180 
181  /* Implicit and explicit semantics - MEM */
182  this->collectNodes(inst.getStoreAccess());
183 
184  /* Implicit and explicit semantics - REG */
185  this->collectNodes(inst.getWrittenRegisters());
186 
187  /* Symbolic Expressions */
188  this->removeSymbolicExpressions(inst);
189  }
190 
191  this->astCtxt->garbage();
192  }
