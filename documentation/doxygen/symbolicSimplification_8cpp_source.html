175  triton::ast::SharedAbstractNode SymbolicSimplification::simplify(const triton::ast::SharedAbstractNode& node) const {
176  std::list<triton::ast::SharedAbstractNode> worklist;
177  triton::ast::SharedAbstractNode snode = node;
178 
179  if (node == nullptr)
180  throw triton::exceptions::SymbolicSimplification("SymbolicSimplification::simplify(): node cannot be null.");
181 
182  if (this->callbacks && this->callbacks->isDefined(triton::callbacks::SYMBOLIC_SIMPLIFICATION)) {
183  snode = this->callbacks->processCallbacks(triton::callbacks::SYMBOLIC_SIMPLIFICATION, node);
184  /*
185  * We use a worklist strategy to avoid recursive calls
186  * and so stack overflow when going through a big AST.
187  */
188  worklist.push_back(snode);
189  while (worklist.size()) {
190  auto ast = worklist.front();
191  worklist.pop_front();
192  bool needs_update = false;
193  for (triton::uint32 index = 0; index < ast->getChildren().size(); index++) {
194  auto child = ast->getChildren()[index];
195  /* Don't apply simplification on nodes like String, Integer, etc. */
196  if (child->getBitvectorSize()) {
197  auto schild = this->callbacks->processCallbacks(triton::callbacks::SYMBOLIC_SIMPLIFICATION, child);
198  ast->setChild(index, schild);
199  needs_update |= !schild->canReplaceNodeWithoutUpdate(child);
200  worklist.push_back(schild);
201  }
202  }
203  if (needs_update) {
204  ast->init(true);
205  }
206  }
207  }
208 
209  return snode;
210  }
