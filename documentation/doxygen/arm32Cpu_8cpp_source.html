244  void Arm32Cpu::disassembly(triton::arch::Instruction& inst) {
245  triton::extlibs::capstone::csh handle;
246  triton::extlibs::capstone::cs_insn* insn;
247  triton::usize count = 0;
248 
249  /* Check if the opcode and opcode' size are defined */
250  if (inst.getOpcode() == nullptr || inst.getSize() == 0)
251  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Opcode and opcodeSize must be definied.");
252 
253  /* Select capstone handler (based on execution mode) */
254  handle = (this->thumb ? this->handleThumb : this->handleArm);
255 
256  /* Clear instructicon's operands if alredy defined */
257  inst.operands.clear();
258 
259  /* Update instruction address if undefined */
260  if (!inst.getAddress()) {
261  inst.setAddress(this->getConcreteRegisterValue(this->getProgramCounter()).convert_to<triton::uint64>());
262  }
263 
264  /* Let's disass and build our operands */
265  count = triton::extlibs::capstone::cs_disasm(handle, inst.getOpcode(), inst.getSize(), inst.getAddress(), 0, &insn);
266  if (count > 0) {
267  triton::extlibs::capstone::cs_detail* detail = insn->detail;
268  for (triton::uint32 j = 0; j < 1; j++) {
269  /* Refine the opcode */
270  inst.setOpcode(insn[j].bytes, insn[j].size);
271 
272  /* Refine the size */
273  inst.setSize(insn[j].size);
274 
275  /* Init the instruction's type */
276  inst.setType(this->capstoneInstructionToTritonInstruction(insn[j].id));
277 
278  /* Init the instruction's code codition */
279  inst.setCodeCondition(this->capstoneConditionToTritonCondition(detail->arm.cc));
280 
281  /* Init the instruction's write back flag */
282  inst.setWriteBack(detail->arm.writeback);
283 
284  /* Set True if the instruction udpate flags */
285  inst.setUpdateFlag(detail->arm.update_flags);
286 
287  /* Set thumb mode */
288  inst.setThumb(thumb);
289 
290  /* Init the disassembly */
291  std::stringstream str;
292 
293  /* Add mnemonic */
294  str << insn[j].mnemonic;
295 
296  /* Add operands */
297  if (inst.getType() == ID_INS_IT || detail->arm.op_count)
298  str << " " << insn[j].op_str;
299 
300  inst.setDisassembly(str.str());
301 
302  /* Process IT instruction */
303  if (inst.getType() == ID_INS_IT) {
304  /* Nested IT instruction, throw an exception as this is not valid ARM code */
305  if (this->itInstrsCount > 0)
306  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Nested IT instructions are not allowed.");
307 
308  /* Copy state from the mnemonic of the instruction */
309  strncpy(this->itStateArray, &insn[j].mnemonic[1], 5);
310  this->itStateArray[4] = 0;
311 
312  this->itInstrsCount = strlen(this->itStateArray);
313  this->itInstrIndex = 0;
314 
315  this->itCC = inst.getCodeCondition();
316  this->itCCInv = this->invertCodeCondition(this->itCC);
317  }
318 
319  /* Process instruction within an IT block */
320  if (inst.getType() != ID_INS_IT && this->itInstrsCount > 0) {
321  /* NOTE Assuming that CS always returns mnemonics in lower case */
322  triton::arch::arm::condition_e cc = this->itStateArray[this->itInstrIndex] == 't' ? this->itCC : this->itCCInv;
323 
324  inst.setCodeCondition(cc);
325 
326  this->itInstrsCount--;
327  this->itInstrIndex++;
328  }
329 
330  /* Init operands */
331  for (triton::uint32 n = 0; n < detail->arm.op_count; n++) {
332  triton::extlibs::capstone::cs_arm_op* op = &(detail->arm.operands[n]);
333  switch(op->type) {
334 
335  case triton::extlibs::capstone::ARM_OP_IMM: {
336  triton::arch::Immediate imm(op->imm, triton::size::dword);
337 
338  if (op->subtracted)
339  imm.setSubtracted(true);
340 
341  inst.operands.push_back(triton::arch::OperandWrapper(imm));
342  break;
343  }
344 
345  case triton::extlibs::capstone::ARM_OP_MEM: {
346  triton::arch::MemoryAccess mem;
347 
348  /* Set the size of the memory access */
349  mem.setBits(triton::bitsize::dword-1, 0);
350 
351  /* LEA if exists */
352  const triton::arch::Register base(*this, this->capstoneRegisterToTritonRegister(op->mem.base));
353  triton::arch::Register index(*this, this->capstoneRegisterToTritonRegister(op->mem.index));
354 
355  /* Set Shift type and value */
356  triton::arch::arm::shift_e shiftType = this->capstoneShiftToTritonShift(op->shift.type);
357 
358  index.setShiftType(shiftType);
359 
360  switch(shiftType) {
361  case triton::arch::arm::ID_SHIFT_INVALID:
362  break;
363  case triton::arch::arm::ID_SHIFT_ASR:
364  case triton::arch::arm::ID_SHIFT_LSL:
365  case triton::arch::arm::ID_SHIFT_LSR:
366  case triton::arch::arm::ID_SHIFT_ROR:
367  index.setShiftValue(op->shift.value);
368  break;
369  case triton::arch::arm::ID_SHIFT_RRX:
370  /* NOTE: According to the manual RRX there is no
371  * immediate associated with this shift type. However,
372  * from the description of the instruction it can be
373  * deduced that a value of one is used.
374  */
375  index.setShiftValue(1);
376  break;
377  case triton::arch::arm::ID_SHIFT_ASR_REG:
378  case triton::arch::arm::ID_SHIFT_LSL_REG:
379  case triton::arch::arm::ID_SHIFT_LSR_REG:
380  case triton::arch::arm::ID_SHIFT_ROR_REG:
381  index.setShiftValue(this->capstoneRegisterToTritonRegister(op->shift.value));
382  break;
383  case triton::arch::arm::ID_SHIFT_RRX_REG:
384  /* NOTE: Capstone considers this as a viable shift operand
385  * but according to the ARM manual this is not possible.
386  */
387  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
388  break;
389  default:
390  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
391  }
392 
393  if (op->subtracted) {
394  index.setSubtracted(true);
395  }
396 
397  triton::uint32 immsize = (
398  this->isRegisterValid(base.getId()) ? base.getSize() :
399  this->isRegisterValid(index.getId()) ? index.getSize() :
400  this->gprSize()
401  );
402 
403  triton::arch::Immediate disp(op->mem.disp, immsize);
404 
405  /* Specify that LEA contains a PC relative */
406  if (base.getId() == this->pcId) {
407  /* NOTE: PC always points to the address to the current
408  * instruction plus: a) 8 in case of ARM mode, or b) 4 in
409  * case of Thumb. It is also aligned to 4 bytes. For more
410  * information, refer to section "Use of labels in UAL
411  * instruction syntax" of the reference manual.
412  */
413  auto offset = this->thumb ? 4 : 8;
414  auto address = (inst.getAddress() + offset) & 0xfffffffc;
415  mem.setPcRelative(address);
416  }
417 
418  /* Note that in ARM32 there is no segment register and scale value */
419  mem.setBaseRegister(base);
420  mem.setIndexRegister(index);
421  mem.setDisplacement(disp);
422 
423  /* If there is an index register available, set scale to 1 to perform this following computation (base) + (index * scale) */
424  if (this->isRegisterValid(index.getId()))
425  mem.setScale(triton::arch::Immediate(1, index.getSize()));
426 
427  inst.operands.push_back(triton::arch::OperandWrapper(mem));
428  break;
429  }
430 
431  case triton::extlibs::capstone::ARM_OP_REG: {
432  triton::arch::Register reg(*this, this->capstoneRegisterToTritonRegister(op->reg));
433 
434  /* Set Shift type and value */
435  triton::arch::arm::shift_e shiftType = this->capstoneShiftToTritonShift(op->shift.type);
436 
437  reg.setShiftType(shiftType);
438 
439  switch(shiftType) {
440  case triton::arch::arm::ID_SHIFT_INVALID:
441  break;
442  case triton::arch::arm::ID_SHIFT_ASR:
443  case triton::arch::arm::ID_SHIFT_LSL:
444  case triton::arch::arm::ID_SHIFT_LSR:
445  case triton::arch::arm::ID_SHIFT_ROR:
446  reg.setShiftValue(op->shift.value);
447  break;
448  case triton::arch::arm::ID_SHIFT_RRX:
449  /* NOTE: According to the manual RRX there is no
450  * immediate associated with this shift type. However,
451  * from the description of the instruction it can be
452  * deduced that a value of one is used.
453  */
454  reg.setShiftValue(1);
455  break;
456  case triton::arch::arm::ID_SHIFT_ASR_REG:
457  case triton::arch::arm::ID_SHIFT_LSL_REG:
458  case triton::arch::arm::ID_SHIFT_LSR_REG:
459  case triton::arch::arm::ID_SHIFT_ROR_REG:
460  reg.setShiftValue(this->capstoneRegisterToTritonRegister(op->shift.value));
461  break;
462  case triton::arch::arm::ID_SHIFT_RRX_REG:
463  /* NOTE: Capstone considers this as a viable shift operand
464  * but according to the ARM manual this is not possible.
465  */
466  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
467  break;
468  default:
469  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
470  }
471 
472  if (op->subtracted)
473  reg.setSubtracted(true);
474 
475  inst.operands.push_back(triton::arch::OperandWrapper(reg));
476  break;
477  }
478 
479  default:
480  /* NOTE: FP, CIMM, and missing one are not supported yet. */
481  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid operand.");
482  } // switch
483  } // for operand
484  } // for instruction
485 
486  /* Set branch */
487  if (detail->groups_count > 0) {
488  for (triton::uint32 n = 0; n < detail->groups_count; n++) {
489  if (detail->groups[n] == triton::extlibs::capstone::ARM_GRP_JUMP) {
490  inst.setBranch(true);
491  inst.setControlFlow(true);
492  }
493  }
494  }
495 
496  /* Post process instruction */
497  this->postDisassembly(inst);
498 
499  /* Free capstone stuffs */
500  triton::extlibs::capstone::cs_free(insn, count);
501  }
502  else
503  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Failed to disassemble the given code.");
504 
505  return;
506  }

509  void Arm32Cpu::postDisassembly(triton::arch::Instruction& inst) const {
510  /* Fix update flag */
511  /* NOTE: Quick (and super ugly) hack. Capstone is reporting
512  * update_flags equals true for ADC, RSC and SBC instruction when
513  * it shouldn't (it should only report true when the S suffix is
514  * present). This will be removed once Capstone is fixed. For more
515  * information, see: https://github.com/aquynh/capstone/issues/1568.
516  */
517  switch (inst.getType()) {
518  case ID_INS_ADC:
519  case ID_INS_RSC:
520  case ID_INS_SBC:
521  if (inst.getDisassembly().at(3) != 's')
522  inst.setUpdateFlag(false);
523  break;
524  }
525 
526  /* Make implicit destination operand explicit */
527  /* NOTE: For some instructions the destination operand is
528  * optional (in which case the first source operand is used as
529  * destination). Capstone returns always all three operands for
530  * ARM instruction (i.e. make the destination operand explicit).
531  * However, it does not do the same for Thumb instructions. Here
532  * we make the destination operand explicit (in order to simplify
533  * the implementation of the semantics).
534  */
535  if (inst.isThumb() && inst.operands.size() == 2) {
536  triton::arch::OperandWrapper op(inst.operands[0]);
537 
538  switch (inst.getType()) {
539  case ID_INS_ADC:
540  case ID_INS_ADD:
541  case ID_INS_AND:
542  case ID_INS_ASR:
543  case ID_INS_BIC:
544  case ID_INS_EOR:
545  case ID_INS_LSL:
546  case ID_INS_LSR:
547  case ID_INS_ORR:
548  case ID_INS_ROR:
549  case ID_INS_SBC:
550  case ID_INS_SUB:
551  inst.operands.insert(inst.operands.begin(), op);
552  break;
553  }
554  }
555 
556  /* NOTE: If the instruction is POP and contains a PC register,
557  * we have to define the instruction as modifiying the control
558  * flow. See #945.
559  */
560  if (inst.getType() == ID_INS_POP) {
561  /* FIXME: Maybe the loop is useless if PC is always the last operand? */
562  for (auto& op : inst.operands) {
563  if (op.getType() == triton::arch::OP_REG && op.getConstRegister().getId() == this->pcId) {
564  inst.setControlFlow(true);
565  break;
566  }
567  }
568  }
569  }

584  triton::uint512 Arm32Cpu::getConcreteMemoryValue(const triton::arch::MemoryAccess& mem, bool execCallbacks) const {
585  triton::uint512 ret = 0;
586  triton::uint64 addr = 0;
587  triton::uint32 size = 0;
588

590  this->callbacks->processCallbacks(triton::callbacks::GET_CONCRETE_MEMORY_VALUE, mem);
591 
592  addr = mem.getAddress();
593  size = mem.getSize();
594 
595  if (size == 0 || size > triton::size::dqqword)
596  throw triton::exceptions::Cpu("Arm32Cpu::getConcreteMemoryValue(): Invalid size memory.");

695  void Arm32Cpu::setConcreteRegisterValue(const triton::arch::Register& reg, const triton::uint512& value) {
696  if (value > reg.getMaxValue())
697  throw triton::exceptions::Register("Arm32Cpu::setConcreteRegisterValue(): You cannot set this concrete value (too big) to this register.");
698 
699  if (this->callbacks)
700  this->callbacks->processCallbacks(triton::callbacks::SET_CONCRETE_REGISTER_VALUE, reg, value);
701 
702  switch (reg.getId()) {
703  case triton::arch::ID_REG_ARM32_R0: (*((triton::uint32*)(this->r0))) = value.convert_to<triton::uint32>(); break;
704  case triton::arch::ID_REG_ARM32_R1: (*((triton::uint32*)(this->r1))) = value.convert_to<triton::uint32>(); break;
705  case triton::arch::ID_REG_ARM32_R2: (*((triton::uint32*)(this->r2))) = value.convert_to<triton::uint32>(); break;
706  case triton::arch::ID_REG_ARM32_R3: (*((triton::uint32*)(this->r3))) = value.convert_to<triton::uint32>(); break;
707  case triton::arch::ID_REG_ARM32_R4: (*((triton::uint32*)(this->r4))) = value.convert_to<triton::uint32>(); break;
708  case triton::arch::ID_REG_ARM32_R5: (*((triton::uint32*)(this->r5))) = value.convert_to<triton::uint32>(); break;
709  case triton::arch::ID_REG_ARM32_R6: (*((triton::uint32*)(this->r6))) = value.convert_to<triton::uint32>(); break;
710  case triton::arch::ID_REG_ARM32_R7: (*((triton::uint32*)(this->r7))) = value.convert_to<triton::uint32>(); break;
711  case triton::arch::ID_REG_ARM32_R8: (*((triton::uint32*)(this->r8))) = value.convert_to<triton::uint32>(); break;
712  case triton::arch::ID_REG_ARM32_R9: (*((triton::uint32*)(this->r9))) = value.convert_to<triton::uint32>(); break;
713  case triton::arch::ID_REG_ARM32_R10: (*((triton::uint32*)(this->r10))) = value.convert_to<triton::uint32>(); break;
714  case triton::arch::ID_REG_ARM32_R11: (*((triton::uint32*)(this->r11))) = value.convert_to<triton::uint32>(); break;
715  case triton::arch::ID_REG_ARM32_R12: (*((triton::uint32*)(this->r12))) = value.convert_to<triton::uint32>(); break;
716  case triton::arch::ID_REG_ARM32_SP: (*((triton::uint32*)(this->sp))) = value.convert_to<triton::uint32>(); break;
717  case triton::arch::ID_REG_ARM32_R14: (*((triton::uint32*)(this->r14))) = value.convert_to<triton::uint32>(); break;
718  case triton::arch::ID_REG_ARM32_PC: {
719  /* NOTE: Once in Thumb mode only switch to ARM through a Branch
720  * and Exchange instruction. The reason for this is that after
721  * switching to Thumb the ISB (instruction set selection bit) is
722  * cleared. Therefore, if we allow to switch back to ARM through
723  * these mechanism we would have a problem processing Thumb
724  * instructions.
725  */
726  auto pc = value.convert_to<triton::uint32>();
727  if (this->isThumb() == false && (pc & 0x1) == 0x1) {
728  this->setThumb(true);
729  }
730  (*((triton::uint32*)(this->pc))) = pc & ~0x1;
731  break;
732  }
733  case triton::arch::ID_REG_ARM32_APSR: (*((triton::uint32*)(this->apsr))) = value.convert_to<triton::uint32>(); break;
734  case triton::arch::ID_REG_ARM32_N: {
735  triton::uint32 b = (*((triton::uint32*)(this->apsr)));
736  (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 31) : b & ~(1 << 31);
737  break;
738  }
739  case triton::arch::ID_REG_ARM32_Z: {
740  triton::uint32 b = (*((triton::uint32*)(this->apsr)));
741  (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 30) : b & ~(1 << 30);
742  break;
743  }
744  case triton::arch::ID_REG_ARM32_C: {
745  triton::uint32 b = (*((triton::uint32*)(this->apsr)));
746  (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 29) : b & ~(1 << 29);
747  break;
748  }
749  case triton::arch::ID_REG_ARM32_V: {
750  triton::uint32 b = (*((triton::uint32*)(this->apsr)));
751  (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 28) : b & ~(1 << 28);
752  break;
753  }
754  default:
755  throw triton::exceptions::Cpu("Arm32Cpu:setConcreteRegisterValue(): Invalid register.");
756  }
