172  /* Removes an aligned memory */
173  void SymbolicEngine::removeAlignedMemory(triton::uint64 address, triton::uint32 size) {
174  /*
175  * Avoid accessing the alignedMemoryReference array when empty. This usually happens when
176  * you initialize the symbolic engine and concretize whole sections of an executable using
177  * setConcreteMemoryValue. No symbolic memory has been created yet but this function will
178  * still try to rougly erase (size * 7) elements.
179  */
180  if (this->alignedMemoryReference.empty())
181  return;
182 
183  /* Remove overloaded positive ranges */
184  for (triton::uint32 index = 0; index < size; index++) {
185  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::byte));
186  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::word));
187  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::dword));
188  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::qword));
189  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::fword));
190  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::dqword));
191  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::qqword));
192  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::dqqword));
193  }
194 
195  /* Remove overloaded negative ranges */
196  for (triton::uint32 index = 1; index < triton::size::dqqword; index++) {
197  if (index < triton::size::word) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::word));
198  if (index < triton::size::dword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::dword));
199  if (index < triton::size::qword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::qword));
200  if (index < triton::size::fword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::fword));
201  if (index < triton::size::dqword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::dqword));
202  if (index < triton::size::qqword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::qqword));
203  if (index < triton::size::dqqword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::dqqword));
204  }
205  }

366  SharedSymbolicExpression SymbolicEngine::newSymbolicExpression(const triton::ast::SharedAbstractNode& node, triton::engines::symbolic::expression_e type, const std::string& comment) {
367  if (this->modes->isModeEnabled(triton::modes::AST_OPTIMIZATIONS)) {
368  /*
369  * Create volatile expression for extended part to avoid long
370  * formulas while printing. Long formulas are introduced by
371  * optimizations of AstContext::concat and AstContext::extract

373  if (node->getType() == triton::ast::ZX_NODE || node->getType() == triton::ast::SX_NODE) {
374  auto n = node->getChildren()[1];
375  if (n->getType() != triton::ast::REFERENCE_NODE && n->getType() != triton::ast::VARIABLE_NODE) {
376  /* FIXME: We lost the origin if we create a new symbolic expression without returning it */
377  auto e = this->newSymbolicExpression(n, VOLATILE_EXPRESSION, "Extended part - " + comment);
378  node->setChild(1, this->astCtxt->reference(e));

547  /* The memory size is used to define the symbolic variable's size. */
548  SharedSymbolicVariable SymbolicEngine::symbolizeMemory(const triton::arch::MemoryAccess& mem, const std::string& symVarAlias) {
549  triton::uint64 memAddr = mem.getAddress();
550  triton::uint32 symVarSize = mem.getSize();
551  triton::uint512 cv = this->architecture->getConcreteMemoryValue(mem);
552 
553  /* First we create a symbolic variable */
554  const SharedSymbolicVariable& symVar = this->newSymbolicVariable(MEMORY_VARIABLE, memAddr, symVarSize * bitsize::byte, symVarAlias);
555 
556  /* Create the AST node */
557  const triton::ast::SharedAbstractNode& symVarNode = this->astCtxt->variable(symVar);
558 
559  /* Setup the concrete value to the symbolic variable */
560  this->setConcreteVariableValue(symVar, cv);
561 
562  /* Record the aligned symbolic variable for a symbolic optimization */
563  if (this->modes->isModeEnabled(triton::modes::ALIGNED_MEMORY)) {
564  const SharedSymbolicExpression& se = this->newSymbolicExpression(symVarNode, MEMORY_EXPRESSION, "aligned Byte reference");
565  se->setOriginMemory(mem);
566  this->addAlignedMemory(memAddr, symVarSize, se);
567  }
568 
569  /* Split expression in bytes */
570  for (triton::sint32 index = symVarSize-1; index >= 0; index--) {
571  triton::uint32 high = ((bitsize::byte * (index + 1)) - 1);
572  triton::uint32 low = ((bitsize::byte * (index + 1)) - bitsize::byte);
573 
574  /* Isolate the good part of the symbolic variable */

576 
577  /* Create a new symbolic expression containing the symbolic variable */
578  const SharedSymbolicExpression& se = this->newSymbolicExpression(tmp, MEMORY_EXPRESSION, "Byte reference");

580 
581  /* Assign the symbolic expression to the memory cell */
582  this->addMemoryReference(memAddr+index, se);
583  }

613  this->assignSymbolicExpressionToRegister(se, parent);

806  /* Returns the AST corresponding to the memory */
807  triton::ast::SharedAbstractNode SymbolicEngine::getMemoryAst(const triton::arch::MemoryAccess& mem) {
808  std::vector<triton::ast::SharedAbstractNode> opVec;
809 
810  triton::ast::SharedAbstractNode tmp = nullptr;
811  triton::uint64 address = mem.getAddress();
812  triton::uint32 size = mem.getSize();
813  triton::uint8 concreteValue[triton::size::dqqword] = {0};
814  triton::uint512 value = this->architecture->getConcreteMemoryValue(mem);
815 
816  triton::utils::fromUintToBuffer(value, concreteValue);
817 
818  /*
819  * Symbolic optimization
820  * If the memory access is aligned, don't split the memory.
821  */
822  if (this->modes->isModeEnabled(triton::modes::ALIGNED_MEMORY) && this->isAlignedMemory(address, size)) {
823  return this->getAlignedMemory(address, size)->getAst();
824  }
825 
826  /* If the memory access is 1 byte long, just return the appropriate 8-bit vector */
827  if (size == 1) {
828  const SharedSymbolicExpression& symMem = this->getSymbolicMemory(address);
829  if (symMem) return this->astCtxt->reference(symMem);
830  else return this->astCtxt->bv(concreteValue[size - 1], bitsize::byte);
831  }
832 
833  /* If the memory access is more than 1 byte long, concatenate each memory cell */
834  opVec.reserve(size);
835  while (size) {
836  const SharedSymbolicExpression& symMem = this->getSymbolicMemory(address + size - 1);
837  if (symMem) opVec.push_back(this->astCtxt->reference(symMem));
838  else opVec.push_back(this->astCtxt->bv(concreteValue[size - 1], bitsize::byte));
839  size--;
840  }
841  return this->astCtxt->concat(opVec);
842  }

905  /* Returns the new symbolic memory expression */
906  const SharedSymbolicExpression& SymbolicEngine::createSymbolicMemoryExpression(triton::arch::Instruction& inst, const triton::ast::SharedAbstractNode& node, const triton::arch::MemoryAccess& mem, const std::string& comment) {
907  std::vector<triton::ast::SharedAbstractNode> ret;
908  triton::ast::SharedAbstractNode tmp = nullptr;
909  SharedSymbolicExpression se = nullptr;
910  triton::uint64 address = mem.getAddress();
911  triton::uint32 writeSize = mem.getSize();
912  triton::usize id = this->uniqueSymExprId;
913 
914  /* Record the aligned memory for a symbolic optimization */
915  if (this->modes->isModeEnabled(triton::modes::ALIGNED_MEMORY)) {
916  const SharedSymbolicExpression& aligned = this->newSymbolicExpression(node, MEMORY_EXPRESSION, "Aligned Byte reference - " + comment);
917  this->addAlignedMemory(address, writeSize, aligned);
918  }
919 
920  /*
921  * As the x86's memory can be accessed without alignment, each byte of the
922  * memory must be assigned to an unique reference.
923  */
924  ret.reserve(mem.getSize());
925  while (writeSize) {
926  triton::uint32 high = ((writeSize * bitsize::byte) - 1);
927  triton::uint32 low = ((writeSize * bitsize::byte) - bitsize::byte);
928  /* Extract each byte of the memory */
929  tmp = this->astCtxt->extract(high, low, node);
930  /* Assign each byte to a new symbolic expression */
931  se = this->newSymbolicExpression(tmp, MEMORY_EXPRESSION, "Byte reference - " + comment);
932  /* Set the origin of the symbolic expression */
933  se->setOriginMemory(triton::arch::MemoryAccess(((address + writeSize) - 1), triton::size::byte));
934  /* ret is the for the final expression */
935  ret.push_back(tmp);
936  /* Assign memory with little endian */
937  this->addMemoryReference((address + writeSize) - 1, se);
938  /* continue */
939  writeSize--;
940  }
941 
942  /* Set implicit read of the base and index registers from an effective address */
943  this->setImplicitReadRegisterFromEffectiveAddress(inst, mem);
944 
945  /* Set explicit write of the memory access */
946  inst.setStoreAccess(mem, node);
947 
948  /* If there is only one reference, we return the symbolic expression */
949  if (ret.size() == 1) {
950  /* Synchronize the concrete state */
951  this->architecture->setConcreteMemoryValue(mem, tmp->evaluate());
952  return this->addSymbolicExpressions(inst, id);
953  }
954 
955  /* Otherwise, we return the concatenation of all symbolic expressions */
956  tmp = this->astCtxt->concat(ret);
957 
958  /* Synchronize the concrete state */
959  this->architecture->setConcreteMemoryValue(mem, tmp->evaluate());
960 
961  se = this->newSymbolicExpression(tmp, MEMORY_EXPRESSION, "Temporary concatenation reference - " + comment);
962  se->setOriginMemory(triton::arch::MemoryAccess(address, mem.getSize()));
963 
964  return this->addSymbolicExpressions(inst, id);
965  }

968  /* Returns the parent AST after inserting the subregister (node) in its AST. */
969  triton::ast::SharedAbstractNode SymbolicEngine::insertSubRegisterInParent(const triton::arch::Register& reg, const triton::ast::SharedAbstractNode& node, bool zxForAssign) {
970  const triton::arch::Register& parentReg = this->architecture->getParentRegister(reg);
971 
972  /* If it's a flag register, there is nothing to do with sub register */
973  if (this->architecture->isFlag(reg)) {
974  return node;
975  }
976 
977  switch (reg.getSize()) {
978  /* ----------------------------------------------------------------*/
979  case triton::size::byte: {
980  const auto& origReg = this->getRegisterAst(parentReg);
981  /*
982  * Mainly used for x86
983  * r[........xxxxxxxx]
984  */
985  if (reg.getLow() == 0) {
986  const auto& keep = this->astCtxt->extract((parentReg.getBitSize() - 1), bitsize::byte, origReg);
987  return this->astCtxt->concat(keep, node);
988  }
989  /*
990  * Mainly used for x86
991  * r[xxxxxxxx........]
992  */
993  else {
994  const auto& keep = this->astCtxt->extract((parentReg.getBitSize() - 1), bitsize::word, origReg);
995  return this->astCtxt->concat(keep, this->astCtxt->concat(node, this->astCtxt->extract((bitsize::byte - 1), 0, origReg)));
996  }
997  }
998 
999  /* ----------------------------------------------------------------*/
1000  case triton::size::word: {
1001  const auto& origReg = this->getRegisterAst(parentReg);
1002  return this->astCtxt->concat(this->astCtxt->extract((parentReg.getBitSize() - 1), bitsize::word, origReg), node);
1003  }
1004 
1005  /* ----------------------------------------------------------------*/
1006  case triton::size::dword:
1007  case triton::size::qword:
1008  case triton::size::fword:
1009  case triton::size::dqword:
1010  case triton::size::qqword:
1011  case triton::size::dqqword: {
1012  if (zxForAssign == false) {
1013  if (parentReg.getBitSize() > reg.getBitSize()) {
1014  const auto& origReg = this->getRegisterAst(parentReg);
1015  return this->astCtxt->concat(this->astCtxt->extract((parentReg.getBitSize() - 1), reg.getHigh() + 1, origReg), node);
1016  }
1017  else {
1018  return node;
1019  }
1020  }
1021  /* zxForAssign == true */
1022  else {
1023  return this->astCtxt->zx(parentReg.getBitSize() - node->getBitvectorSize(), node);
1024  }
1025  }
1026  /* ----------------------------------------------------------------*/
1027  }
1028 
1029  throw triton::exceptions::SymbolicEngine("SymbolicEngine::insertSubRegisterInParent(): Invalid register size.");
1030  }

1096  }
