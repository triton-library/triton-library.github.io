46  SharedAbstractNode AstContext::collect(const SharedAbstractNode& node) {
47  /*
48  * We keep references to nodes which belong to a depth in the AST which is
49  * a multiple of 10000. Thus, when the root node is destroyed, the stack recursivity
50  * stops when the depth level of 10000 is reached, because the nodes there still
51  * have a reference to them in the AST manager. The destruction will continue at the
52  * next allocation of nodes and so on. So, it means that ASTs are destroyed by steps
53  * of depth of 10000 which avoids the overflow while keeping a good scale.
54  *
55  * See: #753.
56  */
57  triton::uint32 lvl = node->getLevel();
58  if (lvl != 0 && (lvl % 10000) == 0) {
59  this->nodes.push_front(node);
60  }
61  return node;
62  }

382  SharedAbstractNode AstContext::bvrol(const SharedAbstractNode& expr, const SharedAbstractNode& rot) {
383  /*
384  * If the mode SYMBOLIZE_INDEX_ROTATION we apply a AST transformation
385  * in order to make index rotation symbolic. Note that this mode increases the
386  * complexity of solving.
387  *
388  * bvrol(expr, rot) = ((expr << (rot % size)) | (expr >> (size - (rot % size))))
389  **/
390  if (this->modes->isModeEnabled(triton::modes::SYMBOLIZE_INDEX_ROTATION)) {
391  auto size = expr->getBitvectorSize();
392  auto bvsize = this->bv(size, size);
393  const auto& node = this->bvor(
394  this->bvshl(expr, this->bvsmod(rot, bvsize)),
395  this->bvlshr(expr, this->bvsub(bvsize, this->bvsmod(rot, bvsize)))
396  );
397  return node;
398  }
399

407  if (node->isSymbolized() == false) {

1081  SharedAbstractNode AstContext::simplify_concat(std::vector<SharedAbstractNode> exprs) {
1082  /*
1083  * Optimization: concatenate extractions in one if possible. We are
1084  * trying to find out whether this is a concatenation of sequential
1085  * extractions from the same AST. Thus, we can make only one extraction.
1086  *
1087  * (concat ((_ extract 31 24) a) ((_ extract 23 16) a)
1088  * ((_ extract 15 8) a) ((_ extract 7 0) a)) =>
1089  * ((_ extract 31 0) a)
1090  *
1091  * So, we are examining whether we can replace concatenation with one
1092  * extraction ((_ extract high low) ast_ref).
1093  **/
1094  uint32 high = 0; // target extraction upper bound
1095  uint32 low = 1; // target extraction lower bound
1096  SharedAbstractNode ast_ref = 0; // target AST to extract from
1097  SharedAbstractNode ast_node = 0; // ast_ref with unrolled references
1098 
1099  /* Try to join all extractions into one from the right to the left */
1100  while (!exprs.empty()) {
1101  /* Get the right most node */
1102  SharedAbstractNode n = exprs.back();
1103  exprs.pop_back();
1104 
1105  /* Returns the first non referene node encountered */
1106  n = triton::ast::dereference(n);
1107 
1108  if (n->getType() == CONCAT_NODE) {
1109  /* Append concatenation children to the right */
1110  for (const SharedAbstractNode& part : n->getChildren()) {
1111  exprs.push_back(part);
1112  }
1113  continue;
1114  }
1115 
1116  if (n->getType() != EXTRACT_NODE) {
1117  /* We cannot optimize if at least one node is not an extract */
1118  return 0;
1119  }
1120 
1121  /* Get extraction arguments */
1122  const auto& childs = n->getChildren();
1123  triton::uint32 hi = reinterpret_cast<IntegerNode*>(childs[0].get())->getInteger().convert_to<uint32>();
1124  triton::uint32 lo = reinterpret_cast<IntegerNode*>(childs[1].get())->getInteger().convert_to<uint32>();
1125  if (hi < lo) {
1126  return 0;
1127  }
1128  n = childs[2];
1129 
1130  /* Returns the first non referene node encountered */
1131  n = triton::ast::dereference(n);
1132 
1133  if (!ast_ref) {
1134  /* First found extraction node */
1135  high = hi;
1136  low = lo;
1137  ast_ref = childs[2];
1138  ast_node = n;
1139  continue;
1140  }
1141 
1142  /*
1143  * Check that unrolled target AST is equal to extraction node child.
1144  * Also, check that extraction lower bound connects with target
1145  * extraction upper bound.
1146  **/
1147  if (high + 1 != lo || !n->equalTo(ast_node)) {
1148  return 0;
1149  }
1150  high = hi;
1151  }
1152 
1153  if (high < low) {
1154  return 0;
1155  }
1156 
1157  /* Perform target extraction */
1158  return this->extract(high, low, ast_ref);
1159  }

1162  SharedAbstractNode AstContext::simplify_extract(triton::uint32 high, triton::uint32 low, const SharedAbstractNode& expr) {
1163  triton::uint32 size = expr->getBitvectorSize();
1164 
1165  if (high <= low || high >= size) {
1166  return 0;
1167  }
1168 
1169  SharedAbstractNode node = expr;
1170  while (true) {
1171  /* Returns the first non referene node encountered */
1172  SharedAbstractNode n = triton::ast::dereference(node);
1173 
1174  if (n->getType() == CONCAT_NODE) {
1175  /*
1176  * Optimization: If we extract the full part of concatenation, just
1177  * return the part. We are trying to find a part of concatenation
1178  * that we can extract from. Thus, we can extract from only one part
1179  * of concatenation.
1180  *
1181  * ((_ extract 11 9) (concat (_ bv1 8) (_ bv2 8) (_ bv3 8) (_ bv4 8))) =>
1182  * ((_ extract 3 1) (_ bv3 8))
1183  */
1184  triton::uint32 hi = n->getBitvectorSize() - 1;
1185  bool found = false;
1186  /* Search for part of concatenation we can extract from. Iterate
1187  * from the left to the right. */
1188  for (const SharedAbstractNode& part : n->getChildren()) {
1189  if (hi < high) {
1190  /* Did not find a part we can extract from */
1191  break;
1192  }
1193  triton::uint32 sz = part->getBitvectorSize();
1194  triton::uint32 lo = hi + 1 - sz;
1195  if (hi == high && lo == low) {
1196  /* We are extracting the full part, just return it */
1197  return part;
1198  }
1199  if (hi >= high && lo <= low) {
1200  /* Extract from part: ((_ extract high-lo low-lo) part) */
1201  node = part;
1202  high -= lo;
1203  low -= lo;
1204  found = true;
1205  break;
1206  }
1207  hi -= sz;
1208  }
1209  if (found) {
1210  /* Optimize ((_ extract high low) node) one more time */
1211  continue;
1212  }
1213  }
1214  else if (n->getType() == ZX_NODE || n->getType() == SX_NODE) {
1215  /*
1216  * Optimization: If we extract from the node being extended, just
1217  * return the node
1218  *
1219  * ((_ extract 31 0) ((_ zero_extend 32) (_ bv1 32))) => (_ bv1 32)
1220  *
1221  * ((_ extract 7 0) ((_ sign_extend 32) (_ bv1 32))) =>
1222  * ((_ extract 7 0) (_ bv1 32))
1223  **/
1224  n = n->getChildren()[1];
1225  triton::uint32 sz = n->getBitvectorSize();
1226  if (low == 0 && high + 1 == sz) {
1227  /* Just return the node being extended */
1228  return n;
1229  }
1230  if (high < sz) {
1231  /* Optimize ((_ extract high low) n) one more time */
1232  node = n;
1233  continue;
1234  }
1235  }
1236  break;
1237  }
1238 
1239  /* Returns the first non referene node encountered */
1240  SharedAbstractNode n = triton::ast::dereference(node);
1241 
1242  /*
1243  * Optimization: extract from extract is one extract
1244  *
1245  * ((_ extract high low) ((_ extract hi lo) a)) =>
1246  * ((_ extract high+lo low+lo) a)
1247  **/
1248  if (n->getType() == EXTRACT_NODE) {
1249  const auto& childs = n->getChildren();
1250  triton::uint32 hi = reinterpret_cast<IntegerNode*>(childs[0].get())->getInteger().convert_to<uint32>();
1251  triton::uint32 lo = reinterpret_cast<IntegerNode*>(childs[1].get())->getInteger().convert_to<uint32>();
1252  if (lo + high <= hi) {
1253  node = childs[2];
1254  high += lo;
1255  low += lo;
1256  }
1257  }
1258 
1259  return node == expr ? 0 : this->extract(high, low, node);
1260  }
