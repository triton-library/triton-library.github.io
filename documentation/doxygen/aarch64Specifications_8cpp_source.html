21  AArch64Specifications::AArch64Specifications(triton::arch::architecture_e arch) {
22  if (arch != triton::arch::ARCH_AARCH64)
23  throw triton::exceptions::Architecture("AArch64Specifications::AArch64Specifications(): Invalid architecture.");
24 
25  // Fill id2reg and name2id with those available in AArch64 from spec
26  #define REG_SPEC(UPPER_NAME, LOWER_NAME, AARCH64_UPPER, AARCH64_LOWER, AARCH64_PARENT, MUTABLE) \
27  id2reg.emplace(ID_REG_AARCH64_##UPPER_NAME, \
28  triton::arch::Register(triton::arch::ID_REG_AARCH64_##UPPER_NAME, \
29  #LOWER_NAME, \
30  triton::arch::ID_REG_AARCH64_##AARCH64_PARENT, \
31  AARCH64_UPPER, \
32  AARCH64_LOWER, \
33  MUTABLE) \
34  ); \
35  name2id.emplace(#LOWER_NAME, ID_REG_AARCH64_##UPPER_NAME);
36  // Handle register not available in capstone as normal registers
37  #define REG_SPEC_NO_CAPSTONE REG_SPEC
38  #include "triton/aarch64.spec"
39  }

42  triton::arch::register_e AArch64Specifications::capstoneRegisterToTritonRegister(triton::uint32 id) const {
43  triton::arch::register_e tritonId = triton::arch::ID_REG_INVALID;
44 
45  switch (id) {
46  // Convert registers from capstone value to triton value
47  #define REG_SPEC(UPPER_NAME, _1, _2, _3, _4, _5) \
48  case triton::extlibs::capstone::ARM64_REG_##UPPER_NAME: \
49  tritonId = triton::arch::ID_REG_AARCH64_##UPPER_NAME; \
50  break;
51  // Ignore registers not available in capstone
52  #define REG_SPEC_NO_CAPSTONE(_1, _2, _3, _4, _5, _6)
53  #include "triton/aarch64.spec"
54 
55  default:
56  tritonId = triton::arch::ID_REG_INVALID;
57  break;
58  }
59 
60  return tritonId;
