288 std::set<const triton::arch::Register*> AArch64Cpu::getParentRegisters(void) const {
289 std::set<const triton::arch::Register*> ret;
290
291 for (const auto& kv: this->id2reg) {
292 auto regId = kv.first;
293 const auto& reg = kv.second;
294
295 /* Skip Vector and System registers */
296 if (this->isVectorRegister(regId) || this->isSystemRegister(regId))
297 continue;
298
299 /* Add GPR */
300 if (reg.getSize() == this->gprSize())
301 ret.insert(&reg);
302
303 /* Add scalar register */
304 if (this->isScalarRegister(regId) && reg.getSize() == triton::bitsize::dqword)
305 ret.insert(&reg);
306
307 /* Add Flags */
308 else if (this->isFlag(regId))
309 ret.insert(&reg);
310 }
311
312 return ret;
313 }

356 void AArch64Cpu::disassembly(triton::arch::Instruction& inst) {
357 triton::extlibs::capstone::cs_insn* insn;
358 triton::usize count = 0;
359 triton::uint32 size = 0;
360
361 /* Check if the opcode and opcode' size are defined */
362 if (inst.getOpcode() == nullptr || inst.getSize() == 0)
363 throw triton::exceptions::Disassembly("AArch64Cpu::disassembly(): Opcode and opcodeSize must be definied.");
364
365 /* Clear instructicon's operands if alredy defined */
366 inst.operands.clear();
367
368 /* Update instruction address if undefined */
369 if (!inst.getAddress()) {
370 inst.setAddress(static_cast<triton::uint64>(this->getConcreteRegisterValue(this->getProgramCounter())));
371 }
372
373 /* Let's disass and build our operands */
374 count = triton::extlibs::capstone::cs_disasm(this->handle, inst.getOpcode(), inst.getSize(), inst.getAddress(), 0, &insn);
375 if (count > 0) {
376 /* Detail information */
377 triton::extlibs::capstone::cs_detail* detail = insn->detail;
378
379 /* Init the disassembly */
380 std::stringstream str;
381
382 str << insn[0].mnemonic;
383 if (detail->arm64.op_count)
384 str << " " << insn[0].op_str;
385
386 inst.setDisassembly(str.str());
387
388 /* Refine the size */
389 inst.setSize(insn[0].size);
390
391 /* Init the instruction's type */
392 inst.setType(this->capstoneInstructionToTritonInstruction(insn[0].id));
393
394 /* Init the instruction's code codition */
395 inst.setCodeCondition(this->capstoneConditionToTritonCondition(detail->arm64.cc));
396
397 /* Init the instruction's write back flag */
398 inst.setWriteBack(detail->arm64.writeback);
399
400 /* Set True if the instruction udpate flags */
401 inst.setUpdateFlag(detail->arm64.update_flags);
402
403 /* Set architecture */
404 inst.setArchitecture(triton::arch::ARCH_AARCH64);
405
406 /* Init operands */
407 for (triton::uint32 n = 0; n < detail->arm64.op_count; n++) {
408 triton::extlibs::capstone::cs_arm64_op* op = &(detail->arm64.operands[n]);
409 switch(op->type) {
410
411 case triton::extlibs::capstone::ARM64_OP_IMM: {
412 triton::arch::Immediate imm(op->imm, size ? size : triton::size::qword);
413
414 /*
415 * Instruction such that CBZ, CBNZ or TBZ may imply a wrong size.
416 * So, if Triton truncates the value by setting a size less than
417 * the original one, we redefine the size automatically.
418 */
419 if (static_cast<triton::uint64>(op->imm) > imm.getValue()) {
420 imm = Immediate();
421 imm.setValue(op->imm, 0); /* By setting 0 as size, we automatically identify the size of the value */
422 }
423
424 /* Set Shift type and value */
425 imm.setShiftType(this->capstoneShiftToTritonShift(op->shift.type));
426 imm.setShiftValue(op->shift.value);
427
428 inst.operands.push_back(triton::arch::OperandWrapper(imm));
429 break;
430 }
431
432 case triton::extlibs::capstone::ARM64_OP_MEM: {
433 triton::arch::MemoryAccess mem;
434
435 /* Set the size of the memory access */
436 mem.setBits(size ? ((size * triton::bitsize::byte) - 1) : triton::bitsize::qword - 1, 0);
437
438 /* LEA if exists */
439 triton::arch::Register base(*this, this->capstoneRegisterToTritonRegister(op->mem.base));
440 triton::arch::Register index(*this, this->capstoneRegisterToTritonRegister(op->mem.index));
441
442 triton::uint32 immsize = (
443 this->isRegisterValid(base.getId()) ? base.getSize() :
444 this->isRegisterValid(index.getId()) ? index.getSize() :
445 this->gprSize()
446 );
447
448 triton::arch::Immediate disp(op->mem.disp, immsize);
449
450 /* Specify that LEA contains a PC relative */
451 /* FIXME: Valid in ARM64 ? */
452 if (base.getId() == this->pcId) {
453 mem.setPcRelative(inst.getNextAddress());
454 }
455
456 /* Set Shift type and value */
457 index.setShiftType(this->capstoneShiftToTritonShift(op->shift.type));
458 index.setShiftValue(op->shift.value);
459
460 /* Set extend type and size */
461 index.setExtendType(this->capstoneExtendToTritonExtend(op->ext));
462 if (op->ext != triton::extlibs::capstone::ARM64_EXT_INVALID) {
463 index.setExtendedSize(base.getBitSize());
464 }
465
466 /* Note that in ARM64 there is no segment register and scale value */
467 mem.setBaseRegister(base);
468 mem.setIndexRegister(index);
469 mem.setDisplacement(disp);
470
471 /* If there is an index register available, set scale to 1 to perform this following computation (base) + (index * scale) */
472 if (this->isRegisterValid(index.getId())) {
473 mem.setScale(triton::arch::Immediate(1, immsize));
474 }
475
476 inst.operands.push_back(triton::arch::OperandWrapper(mem));
477 break;
478 }
479
480 case triton::extlibs::capstone::ARM64_OP_REG: {
481 triton::arch::Register reg(*this, this->capstoneRegisterToTritonRegister(op->reg));
482
483 /* Set Shift type and value */
484 reg.setShiftType(this->capstoneShiftToTritonShift(op->shift.type));
485 reg.setShiftValue(op->shift.value);
486
487 /* Set extend type and size */
488 reg.setExtendType(this->capstoneExtendToTritonExtend(op->ext));
489 if (op->ext != triton::extlibs::capstone::ARM64_EXT_INVALID) {
490 reg.setExtendedSize(size * triton::bitsize::byte);
491 }
492
493 /* Init the vector arrangement specifier */
494 reg.setVASType(this->capstoneVASToTritonVAS(op->vas));
495
496 /* Init the vector index (-1 if irrelevant) */
497 reg.setVectorIndex(op->vector_index);
498
499 /* Define a base address for next operand */
500 size = this->getMemoryOperandSpecialSize(inst.getType());
501 if (!size) {
502 size = reg.getSize();
503 }
504
505 inst.operands.push_back(triton::arch::OperandWrapper(reg));
506 break;
507 }
508
509 case triton::extlibs::capstone::ARM64_OP_SYS: {
510 triton::arch::Register reg(*this, this->capstoneRegisterToTritonRegister(op->reg));
511
512 /* Define a base address for next operand */
513 size = this->getMemoryOperandSpecialSize(inst.getType());
514 if (!size) {
515 size = reg.getSize();
516 }
517
518 inst.operands.push_back(triton::arch::OperandWrapper(reg));
519 break;
520 }
521
522 default:
523 /* NOTE: FP, CIMM, and missing one are not supported yet. */
524 throw triton::exceptions::Disassembly("AArch64Cpu::disassembly(): Invalid operand.");
525 } // switch
526 } // for operand
527
528 /* Set control flow */
529 if (insn[0].id == triton::extlibs::capstone::ARM64_INS_RET)
530 inst.setControlFlow(true);
531
532 /* Set branch */
533 if (detail->groups_count > 0) {
