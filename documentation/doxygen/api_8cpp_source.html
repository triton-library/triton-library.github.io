549  void API::initEngines(void) {
550  this->checkArchitecture();
551 
552  this->symbolic = new(std::nothrow) triton::engines::symbolic::SymbolicEngine(&this->arch, this->modes, this->astCtxt, &this->callbacks);
553  if (this->symbolic == nullptr)
554  throw triton::exceptions::API("API::initEngines(): Not enough memory.");
555 
556  this->solver = new(std::nothrow) triton::engines::solver::SolverEngine();
557  if (this->solver == nullptr)
558  throw triton::exceptions::API("API::initEngines(): Not enough memory.");
559 
560  this->taint = new(std::nothrow) triton::engines::taint::TaintEngine(this->modes, this->symbolic, *this->getCpuInstance());
561  if (this->taint == nullptr)
562  throw triton::exceptions::API("API::initEngines(): Not enough memory.");
563 
564  this->lifting = new(std::nothrow) triton::engines::lifters::LiftingEngine(this->astCtxt, this->symbolic);
565  if (this->lifting == nullptr)
566  throw triton::exceptions::API("API::initEngines(): Not enough memory.");
567 
568  this->irBuilder = new(std::nothrow) triton::arch::IrBuilder(&this->arch, this->modes, this->astCtxt, this->symbolic, this->taint);
569  if (this->irBuilder == nullptr)
570  throw triton::exceptions::API("API::initEngines(): Not enough memory.");
571 
572  /* Setup registers shortcut */
573  this->registers.init(this->arch.getArchitecture());
574  }
