109  static PyObject* PathConstraint_getBranchConstraints(PyObject* self, PyObject* noarg) {
110  try {
111  PyObject* ret = nullptr;
112  const auto& branches = PyPathConstraint_AsPathConstraint(self)->getBranchConstraints();
113 
114  ret = xPyList_New(branches.size());
115  for (triton::usize index = 0; index != branches.size(); index++) {
116  PyObject* dict = xPyDict_New();
117  xPyDict_SetItem(dict, PyStr_FromString("isTaken"), PyBool_FromLong(std::get<0>(branches[index])));
118  xPyDict_SetItem(dict, PyStr_FromString("srcAddr"), PyLong_FromUint64(std::get<1>(branches[index])));
119  xPyDict_SetItem(dict, PyStr_FromString("dstAddr"), PyLong_FromUint64(std::get<2>(branches[index])));
120  xPyDict_SetItem(dict, PyStr_FromString("constraint"), PyAstNode(std::get<3>(branches[index])));
121  PyList_SetItem(ret, index, dict);
122  }
123 
124  return ret;
125  }
126  catch (const triton::exceptions::Exception& e) {
