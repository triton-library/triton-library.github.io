313  static PyObject* Register_richcompare(PyObject* self, PyObject* other, int op) {
314  PyObject* result = nullptr;
315  triton::uint32 id1 = 0;
316  triton::uint32 id2 = 0;
317 
318  if (!PyRegister_Check(other)) {
319  result = Py_NotImplemented;
320  }
321 
322  else {
323  id1 = PyRegister_AsRegister(self)->getId();
324  id2 = PyRegister_AsRegister(other)->getId();
325 
326  switch (op) {
327  case Py_LT:
328  result = (id1 < id2) ? Py_True : Py_False;
329  break;
330  case Py_LE:
331  result = (id1 <= id2) ? Py_True : Py_False;
332  break;
333  case Py_EQ:
334  result = (id1 == id2) ? Py_True : Py_False;
335  break;
336  case Py_NE:

338  break;
339  case Py_GT:

341  break;
342  case Py_GE:
343  result = (id1 >= id2) ? Py_True : Py_False;
344  break;
