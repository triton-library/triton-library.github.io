411  void x86Cpu::disassembly(triton::arch::Instruction& inst) {
412  triton::extlibs::capstone::cs_insn* insn;
413  triton::usize count = 0;
414 
415  /* Check if the opcode and opcode' size are defined */
416  if (inst.getOpcode() == nullptr || inst.getSize() == 0)
417  throw triton::exceptions::Disassembly("x86Cpu::disassembly(): Opcode and opcodeSize must be definied.");
418 
419  /* Clear instructicon's operands if alredy defined */
420  inst.operands.clear();
421 
422  /* Update instruction address if undefined */
423  if (!inst.getAddress()) {
424  inst.setAddress(static_cast<triton::uint64>(this->getConcreteRegisterValue(this->getProgramCounter())));
425  }
426 
427  /* Let's disass and build our operands */
428  count = triton::extlibs::capstone::cs_disasm(this->handle, inst.getOpcode(), inst.getSize(), inst.getAddress(), 0, &insn);
429  if (count > 0) {
430  /* Detail information */
431  triton::extlibs::capstone::cs_detail* detail = insn->detail;
432 
433  /* Init the disassembly */
434  std::stringstream str;
435 
436  str << insn[0].mnemonic;
437  if (detail->x86.op_count)
438  str << " " << insn[0].op_str;
439 
440  inst.setDisassembly(str.str());
441 
442  /* Refine the size */
443  inst.setSize(insn[0].size);
444 
445  /* Init the instruction's type */
446  inst.setType(this->capstoneInstructionToTritonInstruction(insn[0].id));
447 
448  /* Init the instruction's prefix */
449  inst.setPrefix(this->capstonePrefixToTritonPrefix(detail->x86.prefix[0]));
450 
451  /* Set architecture */
452  inst.setArchitecture(triton::arch::ARCH_X86);
453 
454  /* Init operands */
455  for (triton::uint32 n = 0; n < detail->x86.op_count; n++) {
456  triton::extlibs::capstone::cs_x86_op* op = &(detail->x86.operands[n]);
457  switch(op->type) {
458 
459  case triton::extlibs::capstone::X86_OP_IMM:
460  inst.operands.push_back(triton::arch::OperandWrapper(triton::arch::Immediate(op->imm, op->size)));
461  break;
462 
463  case triton::extlibs::capstone::X86_OP_MEM: {
464  triton::arch::MemoryAccess mem;
465 
466  /* Set the size of the memory access */
467  mem.setBits(((op->size * triton::bitsize::byte) - 1), 0);
468 
469  /* LEA if exists */
470  const triton::arch::Register segment(*this, this->capstoneRegisterToTritonRegister(op->mem.segment));
471  const triton::arch::Register base(*this, this->capstoneRegisterToTritonRegister(op->mem.base));
472  const triton::arch::Register index(*this, this->capstoneRegisterToTritonRegister(op->mem.index));
473 
474  triton::uint32 immsize = (
475  this->isRegisterValid(base.getId()) ? base.getSize() :
476  this->isRegisterValid(index.getId()) ? index.getSize() :
477  this->gprSize()
478  );
479 
480  triton::arch::Immediate disp(op->mem.disp, immsize);
481  triton::arch::Immediate scale(op->mem.scale, immsize);
482 
483  /* Specify that LEA contains a PC relative */
484  if (base.getId() == this->pcId)
485  mem.setPcRelative(inst.getNextAddress());
486 
487  mem.setSegmentRegister(segment);
488  mem.setBaseRegister(base);
489  mem.setIndexRegister(index);
490  mem.setDisplacement(disp);
491  mem.setScale(scale);
492 
493  inst.operands.push_back(triton::arch::OperandWrapper(mem));
494  break;
495  }
496 
497  case triton::extlibs::capstone::X86_OP_REG:
498  inst.operands.push_back(triton::arch::OperandWrapper(triton::arch::Register(*this, this->capstoneRegisterToTritonRegister(op->reg))));
499  break;
500 
501  default:
502  break;
503  }
504 
505  }
506  /* Set branch */
507  if (detail->groups_count > 0) {
